42 struct StableElement {
46  using StableElementT = StableElement<T, Compare>;
47 
49  StableElement(T&& o, std::size_t c) : object_(std::move(o)), order_(c) {}
50 
52  StableElement(const T& o, std::size_t c) : object_(o), order_(c) {}
53 
55  virtual ~StableElement() {}
56 
58  operator T() { return object_; }
59 
61  bool operator<(const StableElementT& rhs) const {
62  /* if element priority is same, compare with order */
63  if (comp_(object_, rhs.object_) == false &&
64  comp_(rhs.object_, object_) == false) {
65  return order_ >= rhs.order_;
66  }
67 
68  /* compare two objects */
69  return comp_(object_, rhs.object_);
70  }

75  std::size_t order_;
77  Compare comp_;

81 template <typename T, typename Compare = std::less<T>>
82 class StablePriorityQueue
83  : public std::priority_queue<StableElement<T, Compare>,
84  std::vector<StableElement<T, Compare>>,
85  std::less<StableElement<T, Compare>>> {
86  using stableT = StableElement<T, Compare>;
87  using std::priority_queue<
88  stableT, std::vector<StableElement<T, Compare>>,
89  std::less<StableElement<T, Compare>>>::priority_queue;
90 
91  public:
93  StablePriorityQueue() {}
95  virtual ~StablePriorityQueue() {}
97  const T& front() { return this->c.front().object_; }
99  const T& top() { return this->c.front().object_; }
101  void push(const T& value) {
102  /* push back and increase order */
103  this->c.push_back(stableT(value, counter_++));
104  std::push_heap(this->c.begin(), this->c.end(), this->comp);
105  }
106 
108  void push(T&& value) {
109  this->c.push_back(stableT(std::move(value), counter_++));
110  std::push_heap(this->c.begin(), this->c.end(), this->comp);
111  }
112 
114  template <class... Args>
115  void emplace(Args&&... args) {
116  /* emplace element */
117  this->c.emplace_back(T(std::forward<Args>(args)...), counter_++);
118  std::push_heap(this->c.begin(), this->c.end(), this->comp);
119  }
120 
122  void pop() {
123  /* pop element */
124  std::pop_heap(this->c.begin(), this->c.end(), this->comp);
125  this->c.pop_back();
126  /* if queue is empty, reset counter, this will avoid counter overflow */
127  if (this->empty()) {
128  counter_ = 0;
129  }
130  }
131 
132  protected:
137  std::size_t counter_ = 0;
138 };

513  bool Wait(std::condition_variable& cond, std::unique_lock<std::mutex>& lock,
514  int timeout, std::function<bool()> wait_cond) {
515  bool ret = false;
516  auto cond_func = [&]() { return need_wakeup_ || wait_cond(); };
517 
518  waiter_number_++;
519  if (timeout > 0) {
520  /* if timeout is set, wait for timeout and return false */
521  ret = cond.wait_for(lock, std::chrono::milliseconds(timeout), cond_func);
522  if (ret == false) {
523  errno = ETIMEDOUT;
524  }
525  } else if (timeout == 0) {
526  /* if wait forever, do wait */
527  cond.wait(lock, cond_func);
528  ret = true;
529  } else {
530  /* do not wait */
531  ret = cond_func();
532  }
533 
534  waiter_number_--;
535  /* if wake by Wakeup */
536  if (need_wakeup_) {
537  /* if all waiters have been woken up */
538  if (waiter_number_ == 0) {
539  need_wakeup_ = false;
540  }
541  errno = EINTR;
542  ret = false;
543  }
544 
545  /* wakeup */
546  return ret;

588  bool PushQueue(std::unique_lock<std::mutex>& lock, const T& elem,
589  int timeout) {
590  /* shutdown or queue has enough space */
591  auto wait_check = [&]() {
592  return shutdown_ == true || capacity_ > queue_.size();
593  };
594 
595  /* if queue already shutdown, return false */
596  if (shutdown_) {
597  errno = ESHUTDOWN;
598  return false;

602  if (Wait(not_full_, lock, timeout, wait_check) == false) {
603  return false;
604  }

607  if (shutdown_) {

609  return false;
610  }

625  size_t PushQueue(std::unique_lock<std::mutex>& lock, Sequence* elems,
626  int timeout = 0, size_t expect_space = 1) {
627  size_t push_num = 0;
628 
629  /* shutdown or queue has enough space */
630  auto wait_check = [&]() {
631  return shutdown_ == true || queue_.size() == 0 ||
632  capacity_ >= expect_space + queue_.size();
633  };
634 
635  /* check if queue has space, or wait. */
636  if (Wait(not_full_, lock, timeout, wait_check) == false) {
637  return false;
638  }
639 
640  for (auto it = elems->begin(); it != elems->end(); it++) {
641  if (PushQueue(lock, *it, timeout) == false) {
642  break;
643  }
644 
645  push_num++;
646 
647  /* when get first element, stop waiting */
648  if (timeout >= 0) {
649  timeout = -1;
650  }
651  }
652 
653  /* remove elems from origin sequence */
654  elems->erase(elems->begin(), elems->begin() + push_num);
655 
656  return push_num;
657  }

667  size_t PushQueueForce(std::unique_lock<std::mutex>& lock, Sequence* elems,
668  bool wait_when_full = false, int timeout = 0) {
669  size_t push_num = 0;
670 
671  /* shutdown or queue is not full*/
672  auto wait_check = [&]() {
673  if (wait_when_full == true) {
674  if (capacity_ >= 1 + queue_.size()) {
675  return true;
676  }
677  }
678 
679  return wait_when_full == false || shutdown_ == true || queue_.size() == 0;
680  };
681 
682  /* check if queue has space, or wait. */
683  if (Wait(not_full_, lock, timeout, wait_check) == false) {
684  return false;
685  }
686 
687  for (auto it = elems->begin(); it != elems->end(); it++) {
688  queue_.emplace(*it);
689  push_num++;
690  }
691 
692  /* remove elems from origin sequence */
693  elems->erase(elems->begin(), elems->begin() + push_num);
694

777  /* wait check has data */
778  auto wait_check = [&]() { return shutdown_ == true || queue_.size() > 0; };

780  /* return false when shutdown */
781  if (shutdown_ == true && queue_.size() == 0) {
782  errno = ESHUTDOWN;
783  return false;
784  }
785 
786  /* check if queue has data, or wait. */
787  if (Wait(not_empty_, lock, timeout, wait_check) == false) {
788  return false;
789  }
790 
791  /* shutdown may be called, return false */
792  if (queue_.size() == 0) {
793  errno = ESHUTDOWN;
