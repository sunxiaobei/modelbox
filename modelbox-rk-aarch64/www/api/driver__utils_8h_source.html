32 Status SubProcessRun(func &&fun, ts &&...params) {
33  auto pid = fork();
34  if (pid == 0) {
35  Status ret = fun(params...);
36  if (!ret) {
37  _exit(0);
38  }
39 
40  _exit(1);
41  }
42 
43  if (pid == -1) {
44  MBLOG_ERROR << "fork subprocess failed";
45  return STATUS_FAULT;
46  }
47 
48  MBLOG_INFO << "wait for subprocess " << pid << " process finished";
49  int status;
50  auto ret = waitpid(pid, &status, 0);
51  if (ret < 0) {
52  auto err_msg =
53  "subprocess run failed, wait error, ret:" + std::to_string(errno) +
54  ", msg: " + StrError(errno);
55  MBLOG_ERROR << err_msg;
56  return {STATUS_FAULT, err_msg};
57  }
58 
59  if (WIFSIGNALED(status)) {
60  auto err_msg = "killed by signal " + WTERMSIG(status);
61  MBLOG_ERROR << err_msg;
62  return {STATUS_FAULT, err_msg};
63  } else if(WIFSTOPPED(status)) {
64  auto err_msg = "stopped by signal " + WSTOPSIG(status);
65  MBLOG_ERROR << err_msg;
66  return {STATUS_FAULT, err_msg};
67  }
68 
69  return STATUS_OK;
70 };
